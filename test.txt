name: Publish Staging Enviorment

on:
  push:
    branches: [main]
    
jobs:
  publish:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v2

      - uses: actions/setup-node@v1
        with:
          node-version: 16.x

      - name: Verify Changed Files
        uses: tj-actions/verify-changed-files@v11
        id: verify-changed-files
        with: 
          files: |
            main.js

      - name: Run File Change.
        if: steps.verify-change-files.outputs.files_changed === 'true'
        run: |
          echo "Changed File"

      - name: Verify Changed Files
        uses: tj-actions/changed-files@v29.0.0
        id: changed-files

      - name: Run File Change.
        if: contains(steps.changed-files.outputs.modified_files, 'main.js')
        run: |
          echo 'main.js was changed'

                - name: Verify Changed files
        uses: tj-actions/verify-changed-files@v11.1
        id: verify-changed-files
        with:
          files: |
            main.js

      - name: Run step only when files change.
        run: |
          echo ${{steps.verify-changed-files.outputs.files_changed == 'true'}}
          echo "Does is have it: ${{contains(steps.verify-changed-files.outputs.changed_files, 'main.js')}}"
          echo "Changed files:
          ${{steps.verify-changed-files.outputs.changed_files}}"
          
      - name: Changed Files
        uses: tj-actions/changed-files@v29.0.0
        id: changed-files

      - name: Run File Change.
        run: |
          echo ${{contains(steps.changed-files.outputs.modified_files,
          'main.js')}}

    - name: Get changed.
        uses: jitterbit/get-changed-files@v1
        id: files-changed
        with: 
          format: space-delimited

      - name: Run get files changed.
        if: contains(steps.files-changed.outputs.all, 'main.js')
        run: |
          echo "All: ${{steps.files-changed.outputs.all}}"

          -------------------------------------------------------

publish:
    runs-on: ubuntu-latest

    - name: Get changed.
      uses: jitterbit/get-changed-files@v1
      id: files-changed
      with: 
        format: space-delimited

    if: contains(steps.files-changed.outputs.all, 'main.js') == 'false'

    steps:
      - uses: actions/checkout@v3
      - uses: actions/setup-node@v1
          
      - name: Check publish
        run: |
          echo "Publish run"

  build:
    runs-on: ubuntu-latest

    - name: Get changed.
      uses: jitterbit/get-changed-files@v1
      id: files-changed
      with: 
        format: space-delimited

    if: contains(steps.files-changed.outputs.all, 'main.js') == 'true'

    steps:
      - uses: actions/checkout@v3
      - uses: actions/setup-node@v1

      - name: Check build
        run: |
          echo "Build run"
