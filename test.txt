name: Publish Staging Enviorment

on:
  push:
    branches: [main]
    
jobs:
  publish:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v2

      - uses: actions/setup-node@v1
        with:
          node-version: 16.x

      - name: Verify Changed Files
        uses: tj-actions/verify-changed-files@v11
        id: verify-changed-files
        with: 
          files: |
            main.js

      - name: Run File Change.
        if: steps.verify-change-files.outputs.files_changed === 'true'
        run: |
          echo "Changed File"

      - name: Verify Changed Files
        uses: tj-actions/changed-files@v29.0.0
        id: changed-files

      - name: Run File Change.
        if: contains(steps.changed-files.outputs.modified_files, 'main.js')
        run: |
          echo 'main.js was changed'
